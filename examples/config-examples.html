Configuration Examples

1. Meta Tags Configuration

<meta name="next-api-key" content="your-api-key-here">
<meta name="next-page-type" content="checkout">
<script src="https://campaign-cart-v2.pages.dev/loader.js"></script>

2. JavaScript Configuration

<script>
  window.nextConfig = {
    apiKey: 'your-api-key-here',
    debug: false,
    paymentConfig: {
      expressCheckout: {
        enabled: true,
        methods: {
          paypal: true,
          applePay: true,
          googlePay: false
        }
      }
    },
    addressConfig: {
      defaultCountry: 'US',
      showCountries: ['US', 'CA', 'GB'],
      dontShowStates: ['AS', 'GU', 'PR', 'VI']
    },
    analytics: {
      enabled: true,
      providers: {
        gtm: {
          enabled: true,
          settings: {
            containerId: 'GTM-XXXXX'
          }
        }
      }
    }
  };
</script>
<script src="https://campaign-cart-v2.pages.dev/loader.js"></script>

3. External Configuration File

<!-- Create a file: https://your-cdn.com/next-config.js -->
window.nextConfig = {
  apiKey: 'your-api-key-here',
  // ... rest of config
};

<!-- Then load with: -->
<script src="https://campaign-cart-v2.pages.dev/loader.js" 
        data-config-url="https://your-cdn.com/next-config.js"></script>

4. Mixed Configuration (Meta + JS)

<!-- Critical config in meta -->
<meta name="next-api-key" content="your-api-key-here">

<!-- Additional config in JS -->
<script>
  window.nextConfig = window.nextConfig || {};
  window.nextConfig.analytics = {
    enabled: true,
    providers: {
      custom: {
        enabled: true,
        settings: {
          endpoint: 'https://your-analytics.com/track'
        }
      }
    }
  };
</script>
<script src="https://campaign-cart-v2.pages.dev/loader.js"></script>
