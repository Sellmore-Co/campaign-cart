## Minimal Setup

The simplest possible implementation:

```html
<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <title>My Store</title>
  
  <!-- Configuration -->
  <meta name="os-api-key" content="YOUR_API_KEY">
  
  <!-- Campaign Cart -->
  <script src="https://rtc2.29next.com/campaign-cart/29next.min.js"></script>
</head>
<body>
  <!-- Add to cart button -->
  <button data-os-action="toggle-item" data-os-package="1">Add Product</button>
  
  <!-- Cart info -->
  <p>Cart: <span data-os-cart-count>0</span> items</p>
  <p>Total: <span data-os-cart-total>$0.00</span></p>
  
  <!-- Checkout -->
  <button os-checkout-payment="combo">Checkout</button>
</body>
</html>
```

      const cart = window.twentyNineNext.getCart();

          card.classList.add('selected');

## WordPress Integration

Example for WordPress sites:

```php
<?php
// functions.php
function add_campaign_cart_scripts() {
    ?>
    <script>
    window.osConfig = {
        apiKey: '<?php echo get_option('campaign_cart_api_key'); ?>',
        debug: <?php echo WP_DEBUG ? 'true' : 'false'; ?>
    };
    </script>
    <script src="https://rtc2.29next.com/campaign-cart/29next.min.js"></script>
    <?php
}
add_action('wp_head', 'add_campaign_cart_scripts');

// Shortcode for product buttons
function campaign_cart_product_shortcode($atts) {
    $atts = shortcode_atts(array(
        'id' => '',
        'profile' => '',
        'text' => 'Add to Cart',
        'class' => 'btn btn-primary'
    ), $atts);
    
    if ($atts['profile']) {
        $data_attr = 'data-os-action="toggle-item" data-os-profile="' . esc_attr($atts['profile']) . '"';
    } else {
        $data_attr = 'data-os-action="toggle-item" data-os-package="' . esc_attr($atts['id']) . '"';
    }
    
    return '<button class="' . esc_attr($atts['class']) . '" ' . $data_attr . '>' . 
           esc_html($atts['text']) . '</button>';
}
add_shortcode('campaign_cart_product', 'campaign_cart_product_shortcode');
?>
```

## React Integration

Basic React component example:

```jsx
import React, { useEffect, useState } from 'react';

// Hook for Campaign Cart
function useCampaignCart() {
  const [cart, setCart] = useState({ items: [], count: 0, total: 0 });
  const [isReady, setIsReady] = useState(false);
  
  useEffect(() => {
    // Wait for Campaign Cart to be ready
    if (window.twentyNineNext) {
      setIsReady(true);
      setCart(window.twentyNineNext.getCart());
    } else {
      const handleReady = () => {
        setIsReady(true);
        setCart(window.twentyNineNext.getCart());
      };
      document.addEventListener('29next.ready', handleReady);
      return () => document.removeEventListener('29next.ready', handleReady);
    }
    
    // Listen for cart updates
    const handleCartUpdate = (event) => {
      setCart(event.detail);
    };
    
    document.addEventListener('cart.updated', handleCartUpdate);
    return () => document.removeEventListener('cart.updated', handleCartUpdate);
  }, []);
  
  return { cart, isReady };
}

// Product component
function Product({ profile, name, image }) {
  const { isReady } = useCampaignCart();
  const [price, setPrice] = useState('$0.00');
  
  useEffect(() => {
    if (isReady && window.twentyNineNext.profiles) {
      window.twentyNineNext.profiles.getProfile(profile).then(profileData => {
        setPrice(profileData.formattedPrice);
      });
    }
  }, [isReady, profile]);
  
  const addToCart = () => {
    if (window.twentyNineNext) {
      window.twentyNineNext.profiles.addToCart(profile);
    }
  };
  
  return (
    <div className="product-card">
      <img src={image} alt={name} />
      <h3>{name}</h3>
      <p className="price">{price}</p>
      <button onClick={addToCart} disabled={!isReady}>
        Add to Cart
      </button>
    </div>
  );
}

// Cart component
function CartSummary() {
  const { cart, isReady } = useCampaignCart();
  
  const checkout = () => {
    if (window.twentyNineNext) {
      window.twentyNineNext.checkout();
    }
  };
  
  if (!isReady) return <div>Loading...</div>;
  
  return (
    <div className="cart-summary">
      <h3>Cart ({cart.count} items)</h3>
      <p>Total: {cart.formattedTotal}</p>
      {cart.count > 0 && (
        <button onClick={checkout}>Checkout</button>
      )}
    </div>
  );
}

// App component
function App() {
  return (
    <div className="app">
      <h1>My Store</h1>
      <CartSummary />
      <div className="products">
        <Product 
          profile="starter-kit" 
          name="Starter Kit" 
          image="/starter-kit.jpg" 
        />
        <Product 
          profile="pro-bundle" 
          name="Pro Bundle" 
          image="/pro-bundle.jpg" 
        />
      </div>
    </div>
  );
}

export default App;
```
